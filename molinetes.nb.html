df = df %>%
  group_by(FECHA, DESDE, HASTA, LINEA, ESTACION) %>%  # evaluar agrupar a nivel linea y quedarse con una nica lnea para el anlisis
  summarize(pasajeros = sum(pax_TOTAL)) %>%
  mutate(ts = dmy_hms(paste0(FECHA," " ,DESDE))) %>%
  ungroup() %>%
  select(-c(FECHA, DESDE, HASTA))  ## ver si dejar esto o no
